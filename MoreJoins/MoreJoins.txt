SELECT PRO.NAME, PRO.PRICE, INV.QUANTITY
FROM PRODUCT PRO
INNER JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID;

SELECT *
FROM PRODUCT PRO
LEFT JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID;

SELECT *
FROM PRODUCT PRO
LEFT JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID
WHERE INV.PRODUCT_ID IS NULL;

SELECT *
FROM PRODUCT PRO
RIGHT JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID;

SELECT *
FROM PRODUCT PRO
RIGHT JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID
WHERE PRO.PRODUCT_ID IS NULL;

SELECT *
FROM PRODUCT PRO
FULL OUTER JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID;

SELECT *
FROM PRODUCT PRO
FULL OUTER JOIN INVENTORY INV 
ON PRO.PRODUCT_ID = INV.PRODUCT_ID
WHERE PRO.PRODUCT_ID IS NULL 
OR INV.PRODUCT_ID IS NULL;
